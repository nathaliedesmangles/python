Listes, chaines et visualisation des donn√©es

import matplotlib.pyplot as plt

# Donn√©es
temperatures = [
    [15, 16, 14, 14, 17, 18, 19],  # Ville A
    [22, 23, 21, 20, 24, 25, 26],  # Ville B
    [5, 7, 6, 6, 8, 9, 7],         # Ville C
    [10, 11, 12, 10, 13, 14, 15]   # Ville D
]

villes = ["Ville A", "Ville B", "Ville C", "Ville D"]
jours = ["Lun", "Mar", "Mer", "Jeu", "Ven", "Sam", "Dim"]

# 1. Affichage des temp√©ratures
for i in range(len(villes)):
    print(villes[i], ":", end=" ")	# end permet d'ajouter un espace au lieu d'un saut de ligne entre les villes
    for j in range(len(temperatures[i])):
        print(temperatures[i][j], end=" ") # end permet d'ajouter un espace au lieu d'un saut de ligne
    print()  # retour √† la ligne

# 2. Min et max
for i in range(len(villes)):
    tmax = max(temperatures[i])
    tmin = min(temperatures[i])
    print(f"La temp√©rature maximale de {villes[i]} est {tmax} ¬∞C")
    print(f"La temp√©rature minimale de {villes[i]} est {tmin} ¬∞C")

# 3. Classification des temp√©ratures
for i in range(len(villes)):
    print()
    print(f"Classification pour {villes[i]} :")
    for j in range(len(temperatures[i])):
        t = temperatures[i][j]
        if t < 10:
            print(t, "=> Froide")
        elif 10 <= t <= 20:
            print(t, "=> Douce")
        else:
            print(t, "=> Chaud")

# 4. Graphique
for i in range(len(villes)):
    plt.plot(jours, temperatures[i], label=villes[i])

plt.title("Temp√©ratures hebdomadaires")
plt.xlabel("Jour")
plt.ylabel("Temp√©rature (¬∞C)")
plt.grid(True)
plt.legend()
plt.savefig("temperatures.png")
plt.show()
