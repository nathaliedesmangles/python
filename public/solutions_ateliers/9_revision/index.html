    print(f"{espece} ‚Üí Moyenne = {infos['moyenne']:.2f}, √âcart-type = {infos['ecart_type']:.2f}, N = {infos['n']}")

import matplotlib.pyplot as plt
from scipy.stats import linregress

plt.figure(figsize=(8, 6))

# Trac√© pour chaque esp√®ce
for espece in df_clean["Esp√®ce"].unique():
    sous_df = df_clean[df_clean["Esp√®ce"] == espece]
    x = sous_df["Temp√©rature"]
    y = sous_df["Taux"]

    # Nuage de points
    plt.scatter(x, y, label=espece)

    # R√©gression
    slope, intercept, *_ = linregress(x, y)
    x_range = np.linspace(x.min(), x.max(), 100)
    y_fit = slope * x_range + intercept
    plt.plot(x_range, y_fit, linestyle="--")

plt.title("Taux de photosynth√®se selon la temp√©rature")
plt.xlabel("Temp√©rature (¬∞C)")
plt.ylabel("Taux (Œºmol CO‚ÇÇ/m¬≤/s)")
plt.legend()
plt.grid(True)
plt.tight_layout()

# Sauvegarde
plt.savefig("photosynthese_plot.png")
plt.show()
