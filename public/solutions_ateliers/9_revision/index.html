import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

# √âtape 1 - Lecture
df = pd.read_csv("croissance_algues.csv")

# V√©rifier les NaN
print(df.isna().sum())
df = df.dropna()

# Conversion NumPy
temperatures = df["Temp√©rature"].to_numpy()
taux = df["Taux"].to_numpy()

# √âtape 2 - Analyse par esp√®ce
donnees_par_espece = {}

for _, ligne in df.iterrows():
    espece = ligne["Esp√®ce"]
    taux_val = ligne["Taux"]
    if espece not in donnees_par_espece:
        donnees_par_espece[espece] = []
    donnees_par_espece[espece].append(taux_val)

resume = {}
for espece, valeurs in donnees_par_espece.items():
    valeurs_np = np.array(valeurs)
    resume[espece] = {
        "nb": len(valeurs),
        "moyenne": np.mean(valeurs_np),
        "ecart_type": np.std(valeurs_np)
    }

print("R√©sum√© par esp√®ce :")
for espece, stats in resume.items():
    print(f"{espece} -> {stats}")

# √âtape 3 - Graphique
plt.figure(figsize=(8,6))

especes = df["Esp√®ce"].unique()
for espece in especes:
    sous_df = df[df["Esp√®ce"] == espece]
    x = sous_df["Temp√©rature"].to_numpy()
    y = sous_df["Taux"].to_numpy()
    plt.scatter(x, y, label=f"{espece} (donn√©es)")

    # R√©gression lin√©aire
    coeffs = np.polyfit(x, y, deg=1)
    y_pred = np.polyval(coeffs, x)
    plt.plot(x, y_pred, linestyle="--", label=f"{espece} (r√©gression)")

plt.title("Taux de croissance vs Temp√©rature")
plt.xlabel("Temp√©rature (¬∞C)")
plt.ylabel("Taux de croissance (mm/jour)")
plt.legend()
plt.grid(True)
plt.tight_layout()
plt.savefig("croissance_regression.png")
plt.show()
