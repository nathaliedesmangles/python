Lecture et √©criture de fichiers de donn√©es

import numpy as np
import pandas as pd

# PARTIE 1 ‚Äî Lecture et nettoyage avec pandas

# Lecture du fichier CSV "eau_riviere.csv"
df = pd.read_csv("eau_riviere.csv")

print()
print("Donn√©es initiales (pandas)")
print(df)

# D√©tection des valeurs manquantes
print()
print("Valeurs manquantes par colonne")
print(df.isna().sum())

# Remplacement des valeurs manquantes par la moyenne de la colonne
df["Temperature"] = df["Temperature"].fillna(df["Temperature"].mean())
df["pH"] = df["pH"].fillna(df["pH"].mean())
df["Oxygene"] = df["Oxygene"].fillna(df["Oxygene"].mean())

# OU pour tout le DataFrame d‚Äôun coup
# df = df.fillna(df.mean(numeric_only=True))

print()
print("Donn√©es apr√®s remplissage des NaN")
print(df)


# Ajout de la colonne 'Qualite' selon la concentration en oxyg√®ne
# R√®gles :
# - Oxyg√®ne >= 8.0 ‚Üí "Bonne"
# - 7.5 <= Oxyg√®ne < 8.0 ‚Üí "Moyenne"
# - Oxyg√®ne < 7.5 ‚Üí "Faible"

df["Qualite"] = np.where(
    df["Oxygene"] >= 8.0, "Bonne",
    np.where(
        (df["Oxygene"] >= 7.5) & (df["Oxygene"] < 8.0), "Moyenne",
        "Faible"
    )
)


# Ajout de la colonne 'Alerte' selon le pH
# Si pH < 7.0 ‚Üí "Attention", sinon "OK"

df["Alerte"] = np.where(df["pH"] < 7.0, "Attention", "OK")

print()
print("Donn√©es finales avec colonnes d√©riv√©es")
print(df)


# Sauvegarde finale du fichier nettoy√©
df.to_csv("eau_riviere_nettoyee.csv", index=False)
print()
print("Fichier 'eau_riviere_nettoyee.csv' sauvegard√© avec succ√®s.")
