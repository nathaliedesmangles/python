import numpy as np
import matplotlib.pyplot as plt

# Donn√©es exp√©rimentales
temps = np.array([0, 1, 2, 3, 4])
distance = np.array([0, 0.9, 1.8, 2.9, 4.1])

# R√©gression lin√©aire (degr√© 1 = droite)
a, b = np.polyfit(temps, distance, 1)

print(f"Pente a = {a:.2f} m/s")
print(f"Ordonn√©e √† l‚Äôorigine b = {b:.2f} m")

# Calcul des valeurs pr√©dites par la droite

# Calcul du coefficient de d√©termination R¬≤
distance_moy = np.mean(distance)

ss_res = np.sum((distance - distance_predite)**2)
r2 = 1 - (ss_res / ss_tot)
print(f"Coefficient de d√©termination R¬≤ = {r2:.3f}")

# Trac√© du graphique (nuage de points) et de la droite
plt.scatter(temps, distance, color='blue', label='Donn√©es exp√©rimentales')
plt.plot(temps, distance_predite, color='red', label='Droite de r√©gression')

plt.xlabel("Temps (s)")
plt.ylabel("Distance (m)")
plt.title("Mouvement rectiligne uniforme - R√©gression lin√©aire")
plt.legend()
plt.grid(True)
plt.show()
