Exercice 1 ‚Äì Chute libre verticale et r√©gression lin√©aire

import numpy as np
import matplotlib.pyplot as plt

# Constantes
g = 9.8
h = 20

# Tableau de temps
t = np.linspace(0, 2, 21)

# Position
y = h - 0.5 * g * t**2

print("5 premi√®res valeurs de y :", y[:5])

# R√©gression lin√©aire entre 0.5s et 1.5s
masque = (t >= 0.5) & (t <= 1.5)
pente, ordonnee = np.polyfit(t[masque], y[masque], 1)  # degr√© 1
droite = pente * t + ordonnee

# Graphique
plt.plot(t, y, 'bo-', label="Chute libre")
plt.plot(t, droite, 'r--', label="R√©gression lin√©aire (0.5s-1.5s)")
plt.xlabel("Temps (s)")
plt.ylabel("Position (m)")
plt.legend()
plt.title("Chute libre verticale avec r√©gression locale")
plt.show()

Exercice 2 ‚Äì Mouvement rectiligne uniforme et r√©gression lin√©aire

import numpy as np
import matplotlib.pyplot as plt

# Vitesse constante
v = 15  

# Temps de 0 √† 10 s
t = np.arange(0, 10.5, 0.5)

# Position
x = v * t

print("Derni√®re position :", x[-1])

# R√©gression lin√©aire
a, b = np.polyfit(t, x, 1)
droite = a * t + b

# Graphique
plt.plot(t, x, 'bo', label="Position calcul√©e")
plt.plot(t, droite, 'r--', label="R√©gression lin√©aire")
plt.xlabel("Temps (s)")
plt.ylabel("Position (m)")
plt.title("Mouvement rectiligne uniforme")
plt.legend()
plt.show()

Exercice 3 ‚Äì √ânergie cin√©tique et barres d’erreur

import numpy as np
import matplotlib.pyplot as plt

# Masse
m = 2.0

# Tableau des vitesses

# √ânergie cin√©tique
Ec = 0.5 * m * v**2

print("√ânergies cin√©tiques :", Ec)
print("√ânergie maximale :", np.max(Ec))
print("Trois derni√®res valeurs :", Ec[-3:])
print("Vitesses 2 √† 8 m/s :", v[1:5])
print("√ânergies correspondantes :", Ec[1:5])

# Barres d‚Äôerreur (5 % de la valeur)
erreurs = 0.05 * Ec  

# Graphique √† barres
plt.bar(v, Ec, yerr=erreurs, capsize=5, color='skyblue', edgecolor='black')
plt.xlabel("Vitesse (m/s)")
plt.ylabel("√ânergie cin√©tique (J)")
plt.title("√ânergie cin√©tique en fonction de la vitesse")
plt.show()
